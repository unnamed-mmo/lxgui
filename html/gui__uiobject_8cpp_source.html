778 void uiobject::read_anchors_(float& iLeft, float& iRight, float& iTop,
779  float& iBottom, float& iXCenter, float& iYCenter) const
780 {
781  iLeft = +std::numeric_limits<float>::infinity();
782  iRight = -std::numeric_limits<float>::infinity();
783  iTop = +std::numeric_limits<float>::infinity();
784  iBottom = -std::numeric_limits<float>::infinity();
785 
786  for (const auto& mOptAnchor : lAnchorList_)
787  {
788  if (!mOptAnchor)
789  continue;
790 
791  const anchor& mAnchor = mOptAnchor.value();
792 
793  // Make sure the anchored object has its borders updated
794  const uiobject* pObj = mAnchor.get_parent();
795  if (pObj)
796  pObj->update_borders_();
797 
798  switch (mAnchor.get_point())
799  {
800  case anchor_point::TOPLEFT :
801  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
802  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
803  break;
804  case anchor_point::TOP :
805  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
806  iXCenter = mAnchor.get_abs_x();
807  break;
808  case anchor_point::TOPRIGHT :
809  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
810  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
811  break;
812  case anchor_point::RIGHT :
813  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
814  iYCenter = mAnchor.get_abs_y();
815  break;
816  case anchor_point::BOTTOMRIGHT :
817  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
818  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
819  break;
820  case anchor_point::BOTTOM :
821  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
822  iXCenter = mAnchor.get_abs_x();
823  break;
824  case anchor_point::BOTTOMLEFT :
825  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
826  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
827  break;
828  case anchor_point::LEFT :
829  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
830  iYCenter = mAnchor.get_abs_y();
831  break;
832  case anchor_point::CENTER :
833  iXCenter = mAnchor.get_abs_x();
834  iYCenter = mAnchor.get_abs_y();
835  break;
836  }
837  }

893 void uiobject::update_anchors()
894 {
895  if (!bUpdateAnchors_)
896  return;
897 
898  std::vector<const uiobject*> lAnchorParentList;
899  for (auto& mAnchor : lAnchorList_)
900  {
901  if (!mAnchor)
902  continue;
903 
904  const uiobject* pObj = mAnchor->get_parent();
905  if (pObj)
906  {
907  if (pObj->depends_on(this))
908  {
909  gui::out << gui::error << "gui::" << lType_.back() << " : Cyclic anchor dependency ! "
910  << "\"" << sName_ << "\" and \"" << pObj->get_name() << "\" depend on "
911  "eachothers (directly or indirectly).\n\""
912  << anchor::get_string_point(mAnchor->get_point()) << "\" anchor removed." << std::endl;
913 
914  mAnchor.reset();
915  continue;
916  }
917 
918  if (utils::find(lAnchorParentList, pObj) == lAnchorParentList.end())
919  lAnchorParentList.push_back(pObj);
920  }
921  }
922 
923  for (const auto* pParent : lPreviousAnchorParentList_)
924  {
925  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
926  pParent->notify_anchored_object(this, false);
927  }
928 
929  for (const auto* pParent : lAnchorParentList)
930  {
931  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
932  pParent->notify_anchored_object(this, true);
933  }
934 
935  lPreviousAnchorParentList_ = std::move(lAnchorParentList);
936  bUpdateAnchors_ = false;
937 }
