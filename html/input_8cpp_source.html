333 void manager::update(float fTempDelta)
334 {
335  if (bRemoveKeyboardFocus_)
336  {
337  bKeyboardFocus_ = false;
338  pKeyboardFocusReceiver_ = nullptr;
339  bRemoveKeyboardFocus_ = false;
340  }
341 
342  if (bRemoveMouseFocus_)
343  {
344  bMouseFocus_ = false;
345  pMouseFocusReceiver_ = nullptr;
346  bRemoveMouseFocus_ = false;
347  }
348 
349  if (!pSource_->is_manually_updated())
350  pSource_->update();
351 
352  // Control extreme delta time after loading/at startup etc
353  double dDelta = fTempDelta;
354  if ((dDelta < 0.0) || (dDelta > 1.0))
355  dDelta = 0.05;
356 
357  // Update keys
358  const source::key_state& mKeyState = pSource_->get_key_state();
359  bKey_ = false;
360  for (uint i = 0; i < KEY_NUMBER; ++i)
361  {
362  // Update delays
363  if (mKeyState.lKeyState[i])
364  {
365  bKey_ = true;
366  lKeyDelay_[i] += dDelta;
367  if (lKeyDelay_[i] >= dLongPressDelay_)
368  lKeyLong_[i] = true;
369  }
370  else
371  {
372  lKeyDelay_[i] = 0.0;
373  lKeyLong_[i] = false;
374  }
375  }
376 
377  // Handle modifier keys
378  bCtrlPressed_ = key_is_down(key::K_LCONTROL, true) || key_is_down(key::K_RCONTROL, true);
379  bShiftPressed_ = key_is_down(key::K_LSHIFT, true) || key_is_down(key::K_RSHIFT, true);
380  bAltPressed_ = key_is_down(key::K_LMENU, true) || key_is_down(key::K_RMENU, true);
381 
382  // Update mouse state
383  const source::mouse_state& mMouseState = pSource_->get_mouse_state();
384  for (uint i = 0; i < MOUSE_BUTTON_NUMBER; ++i)
385  {
386  // Update delays
387  if (mMouseState.lButtonState[i])
388  {
389  lMouseDelay_[i] += dDelta;
390  if (lMouseDelay_[i] >= dLongPressDelay_)
391  lMouseLong_[i] = true;
392  }
393  else
394  {
395  lMouseDelay_[i] = 0.0;
396  lMouseLong_[i] = false;
397  }
398  }
399 
400  // Send events
401  for (auto& mEvent : pSource_->poll_events())
402  {
403  if (mEvent.get_name() == "KEY_PRESSED" || mEvent.get_name() == "KEY_RELEASED")
404  {
405  // Add key name to the event
406  mEvent.add(get_key_name(utils::get<key>(mEvent.get(0))));
407  }
408  else if (mEvent.get_name() == "MOUSE_PRESSED" ||
409  mEvent.get_name() == "MOUSE_RELEASED" ||
410  mEvent.get_name() == "MOUSE_DOUBLE_CLICKED")
411  {
412  // Add button name to the event
413  mEvent.add(get_mouse_button_string(mEvent.get<mouse_button>(0)));
414  }
415 
416  fire_event_(mEvent);
417  }
418 
419  lChars_ = pSource_->get_chars();
420  if (!lChars_.empty())
421  {
422  gui::event mCharEvent("TEXT_ENTERED");
423  mCharEvent.add(std::uint32_t{});
424  for (auto cChar : lChars_)
425  {
426  mCharEvent.get(0) = cChar;
427  fire_event_(mCharEvent);
428  }
429  }
430 
431  // Update mouse position
432  if (mMouseState.bHasDelta)
433  {
434  fDMX_ = mMouseState.fDX/fScalingFactor_;
435  fDMY_ = mMouseState.fDY/fScalingFactor_;
436  fRelDMX_ = mMouseState.fRelDX/fScalingFactor_;
437  fRelDMY_ = mMouseState.fRelDY/fScalingFactor_;
438  }
439  else
440  {
441  fDMX_ = (mMouseState.fAbsX - fMX_)/fScalingFactor_;
442  fDMY_ = (mMouseState.fAbsY - fMY_)/fScalingFactor_;
443  fRelDMX_ = (mMouseState.fRelX - fRelMX_)/fScalingFactor_;
444  fRelDMY_ = (mMouseState.fRelY - fRelMY_)/fScalingFactor_;
445  }
446 
447  fMX_ = mMouseState.fAbsX/fScalingFactor_;
448  fMY_ = mMouseState.fAbsY/fScalingFactor_;
449  fRelMX_ = mMouseState.fRelX/fScalingFactor_;
450  fRelMY_ = mMouseState.fRelY/fScalingFactor_;
451 
452  fMWheel_ = mMouseState.fRelWheel;
453  if (fMWheel_ == 0.0f)
454  bWheelRolled_ = false;
455  else
456  bWheelRolled_ = true;
457 
458  // Send movement event
459  if (fDMX_ != 0.0 || fDMY_ != 0.0)
460  {
461  if (!bMouseDragged_)
462  {
463  uint iMouseButtonPressed = uint(-1);
464  for (uint i = 0; i < MOUSE_BUTTON_NUMBER; ++i)
465  {
466  if (mMouseState.lButtonState[i])
467  {
468  iMouseButtonPressed = i;
469  break;
470  }
471  }
472 
473  if (iMouseButtonPressed != uint(-1))
474  {
475  bMouseDragged_ = true;
476  mMouseDragButton_ = static_cast<mouse_button>(iMouseButtonPressed);
477 
478  gui::event mMouseDragEvent("MOUSE_DRAG_START", true);
479  mMouseDragEvent.add(iMouseButtonPressed);
480  mMouseDragEvent.add(mMouseState.fAbsX);
481  mMouseDragEvent.add(mMouseState.fAbsY);
482  mMouseDragEvent.add(get_mouse_button_string(mMouseDragButton_));
483  fire_event_(mMouseDragEvent, true);
484  }
485  }
486 
487  gui::event mMouseMovedEvent("MOUSE_MOVED", true);
488  mMouseMovedEvent.add(fDMX_);
489  mMouseMovedEvent.add(fDMY_);
490  mMouseMovedEvent.add(fDMX_*mMouseState.fRelX/mMouseState.fAbsX);
491  mMouseMovedEvent.add(fDMY_*mMouseState.fRelY/mMouseState.fAbsY);
492  fire_event_(mMouseMovedEvent, true);
493  }
494 
495  if (bMouseDragged_ && !mMouseState.lButtonState[(uint)mMouseDragButton_])
496  {
497  bMouseDragged_ = false;
498 
499  gui::event mMouseDragEvent("MOUSE_DRAG_STOP", true);
500  mMouseDragEvent.add((uint)mMouseDragButton_);
501  mMouseDragEvent.add(mMouseState.fAbsX);
502  mMouseDragEvent.add(mMouseState.fAbsY);
503  mMouseDragEvent.add(get_mouse_button_string(mMouseDragButton_));
504  fire_event_(mMouseDragEvent, true);
505  }
506 
507  if (bWheelRolled_)
508  {
509  gui::event mMouseWheelEvent("MOUSE_WHEEL", true);
510  mMouseWheelEvent.add(fMWheel_);
511  fire_event_(mMouseWheelEvent, true);
512  }
513 
514  if (pSource_->has_window_resized())
515  {
516  gui::event mWindowResizedEvent("WINDOW_RESIZED", true);
517  mWindowResizedEvent.add(pSource_->get_window_width());
518  mWindowResizedEvent.add(pSource_->get_window_height());
519  fire_event_(mWindowResizedEvent, true);
520  pSource_->reset_window_resized();
521  }
522 
523  dTime_ += dDelta;
524 }
